29template<typename T, int Bits, int Offset>
30class StaticSizedBitTrie {
31public:
32 using Child = StaticSizedBitTrie<T, Bits, Offset - Bits>;
33 using bitmap_t = uint64_t;
34 using size_type = std::size_t;
35 static constexpr int branch_mask = (1 << Bits) - 1;
36
38 StaticSizedBitTrie() : size_(0), bitmap(0) {}
39
41 size_type size() const { return size_; }
42
44 bool empty() const { return size_ == 0; }
45
47 bool insert(T val) {
48 int idx = (val >> Offset) & branch_mask;
49 size_type pos = impl::popcount_low(bitmap, idx);
50 if (!((bitmap >> idx) & 1)) {
51 bitmap |= bitmap_t(1) << idx;
52 children.emplace(children.begin() + pos);
53 }
54 bool ret = children[pos].insert(val);
55 size_ += ret;
56 return ret;
57 }
58
60 bool erase(T val) {
61 int idx = (val >> Offset) & branch_mask;
62 if ((bitmap >> idx) & 1) {
63 size_type pos = impl::popcount_low(bitmap, idx);
64 bool ret = children[pos].erase(val);
65 size_ -= ret;
66 if (children[pos].empty()) {
67 bitmap &= ~(bitmap_t(1) << idx);
68 children.erase(children.begin() + pos);
69 }
70 return ret;
71 } else {
72 return false;
73 }
74 }
75
77 bool find(T val) const {
78 int idx = (val >> Offset) & branch_mask;
79 if ((bitmap >> idx) & 1) {
80 size_type pos = impl::popcount_low(bitmap, idx);
81 return children[pos].find(val);
82 } else {
83 return false;
84 }
85 }
86
94 T xor_min(T xor_val) const {
95 int curr_xor_val = (xor_val >> Offset) & branch_mask;
96 int curr_xor_min = port::countr_zero(impl::xor_permute(bitmap, curr_xor_val));
97 size_type pos = impl::popcount_low(bitmap, curr_xor_min ^ curr_xor_val);
98 return (T(curr_xor_min) << Offset) | children[pos].xor_min(xor_val);
99 }

102 std::optional<T> next(T val) const {
103 int idx = (val >> Offset) & branch_mask;
104 int next_idx = impl::next_set_bit(bitmap, idx);
105 if (next_idx == 64) {
106 return std::nullopt;
107 } else {
108 size_type pos = impl::popcount_low(bitmap, next_idx);
109 if (next_idx > idx) {
110 return (T(next_idx) << Offset) | *(children[pos].next(0));
111 }
112 auto ret = children[pos].next(val);
113 if (ret) {
114 return (T(next_idx) << Offset) | *ret;
115 }
116 next_idx = impl::next_set_bit(bitmap, next_idx + 1);
117 if (next_idx == 64) {
118 return std::nullopt;
119 } else {
120 return (T(next_idx) << Offset) | *(children[pos + 1].next(0));
121 }
122 }
123 }
124
126 std::optional<T> prev(T val) const {
127 int idx = (val >> Offset) & branch_mask;
128 int prev_idx = impl::prev_set_bit(bitmap, idx + 1);
129 if (prev_idx == -1) {
130 return std::nullopt;
131 } else {
132 size_type pos = impl::popcount_low(bitmap, prev_idx);
133 constexpr T MAX = (T(1) << Offset) - 1;
134 if (prev_idx < idx) {
135 return (T(prev_idx) << Offset) | *(children[pos].prev(MAX));
136 }
137 auto ret = children[pos].prev(val);
138 if (ret) {
139 return (T(prev_idx) << Offset) | *ret;
140 }
141 prev_idx = impl::prev_set_bit(bitmap, prev_idx);
142 if (prev_idx == -1) {
143 return std::nullopt;
144 } else {
145 return (T(prev_idx) << Offset) | *(children[pos - 1].prev(MAX));
146 }
147 }
148 }

150private:
151 bitmap_t bitmap;
152 size_type size_;
153 std::vector<Child> children;
154};

157template<typename T, int Bits>
158class StaticSizedBitTrie<T, Bits, 0> {
159public:
160 using bitmap_t = uint64_t;
161 using size_type = std::size_t;
162 static constexpr int branch_mask = (1 << Bits) - 1;
163
164 StaticSizedBitTrie() : bitmap(0) {}
165
166 size_type size() const { return port::popcount(bitmap); }
167
168 bool empty() const { return bitmap == 0; }
169
170 bool insert(T val) {
171 int idx = val & branch_mask;
172 bool ret = !((bitmap >> idx) & 1);
173 bitmap |= bitmap_t(1) << idx;
174 return ret;
175 }
176
177 bool erase(T val) {
178 int idx = val & branch_mask;
179 bool ret = (bitmap >> idx) & 1;
180 bitmap &= ~(bitmap_t(1) << idx);
181 return ret;
182 }
183
184 bool find(T val) const {
185 return (bitmap >> (val & branch_mask)) & 1;
186 }
187
188 T xor_min(T xor_val) const {

190 }

193 int ret = impl::next_set_bit(bitmap, val & branch_mask);
194 return ret == 64 ? std::nullopt : std::make_optional(ret);
195 }

198 int ret = impl::prev_set_bit(bitmap, (val & branch_mask) + 1);
199 return ret == -1 ? std::nullopt : std::make_optional(ret);
200 }
