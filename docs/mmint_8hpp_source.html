13template<typename UInt>
14class MontgomeryReduction {
15public:
16 using int_type = UInt;
17 using int_double_t = make_double_width_t<int_type>;
18 static constexpr int base_width = std::numeric_limits<int_type>::digits;
19
20 constexpr explicit MontgomeryReduction(int_type mod) :
21 _mod(mod),
22 _mod_neg_inv(inv_base(~(mod - 1))),
23 _mbase((int_double_t(1) << base_width) % mod),
24 _mbase2(int_double_t(_mbase) * _mbase % mod),
25 _mbase3(int_double_t(_mbase2) * _mbase % mod)
26 {}
27
28 // N
29 constexpr int_type mod() const { return _mod; }
30
31 // R%N
32 constexpr int_type mbase() const { return _mbase; }
33
34 // R^2%N
35 constexpr int_type mbase2() const { return _mbase2; }
36
37 // R^3%N
38 constexpr int_type mbase3() const { return _mbase3; }
39
40 // T*(R^-1)%N. Result <2N if input <2N*2N.
41 constexpr int_type reduce(int_double_t t) const {
42 int_type m = int_type(t) * _mod_neg_inv;
43 int_type r = (t + int_double_t(m) * _mod) >> base_width;
44 return r;
45 }
46
47 // Shrink value into range [0,2N)
48 constexpr int_type shrink(int_type x) const {
49 return x >= _mod * 2 ? x - _mod * 2 : x;
50 }
51
52 // Shrink value into range [0,N)
53 constexpr int_type strict_shrink(int_type x) const {
54 return x >= _mod ? x - _mod : x;
55 }
56
57private:
58 int_type _mod, _mod_neg_inv, _mbase, _mbase2, _mbase3;
59
60 // Modular inverse modulo 2^2^k by Hensel lifting.
61 static constexpr int_type inv_base(int_type x) {
62 int_type y = 1;
63 for (int i = 1; i < base_width; i *= 2) {
64 y *= int_type(2) - x * y;
65 }
66 return y;
67 }
68};

296 struct ModGuard {
