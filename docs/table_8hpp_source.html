211 void rehash(size_type new_cap) {
212 assert(new_cap >= 4 && (new_cap & (new_cap - 1)) == 0);
213 std::vector<impl::HashCell<T>> old_cells = std::move(cells);
214 size_type old_cap = cap_mask + 1;
