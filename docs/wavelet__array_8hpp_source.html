45 template <typename BidirIt>
46 static WaveletArray build_and_sort(BidirIt first, BidirIt last) {
47 WaveletArray<T, M> wa;
48 std::size_t size = std::distance(first, last);
49 for (int lvl = M - 1; lvl >= 0; lvl--) {
50 BitDictBuilder builder(size);
51 BitDict::size_type i = 0;
52 for (auto it = first; it != last; ++it) {
53 builder.set(i++, (*it >> lvl) & 1);
54 }
55 wa.bit_dict[lvl] = builder.build();
56 stable_partition(first, last, [=](T x) { return !((x >> lvl) & 1); });
57 }
58 return wa;
59 }

84 T range_nth(size_type left, size_type right, size_type n) const {
85 T ret = 0;
86 for (int lvl = M - 1; lvl >= 0; lvl--) {
87 const BitDict &bd = bit_dict[lvl];
88 size_type zero_count = bd.rank0(right) - bd.rank0(left);
89 bool bit = n >= zero_count;
90 ret |= T(bit) << lvl;
91 if (bit) {
92 n -= zero_count;
93 }
94 left = bd.rank_to_child(left, bit);
95 right = bd.rank_to_child(right, bit);
96 }

121 return _rangefreq(left, right, low, high, M - 1);

124private:
125 std::array<BitDict, M> bit_dict;
126
127 size_type _rangefreq(size_type left, size_type right, T low, T high, int lvl) const {
128 if (left >= right) {
129 return 0;
130 } else if (high - low == (lvl == M - 1 ? max_value : (T(1) << (lvl + 1)) - 1)) {
131 return right - left;
132 }
133 T bit_mask = T(1) << lvl;
134 const BitDict &bd = bit_dict[lvl];
135 if (bit_mask & (low ^ high)) {
136 T split = ~(bit_mask - 1) & high;
137 return _rangefreq(bd.rank_to_child(left, false), bd.rank_to_child(right, false), low, split - 1, lvl - 1)
138 + _rangefreq(bd.rank_to_child(left, true), bd.rank_to_child(right, true), split, high, lvl - 1);
139 } else {
140 bool bit = bit_mask & low;
141 return _rangefreq(bd.rank_to_child(left, bit), bd.rank_to_child(right, bit), low, high, lvl - 1);
142 }
143 }
