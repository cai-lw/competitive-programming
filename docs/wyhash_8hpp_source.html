14static inline void _wymum(uint64_t *A, uint64_t *B){
15 __uint128_t r=*A; r*=*B;
16 *A=(uint64_t)r; *B=(uint64_t)(r>>64);
17}

19//multiply and xor mix function, aka MUM
20static inline uint64_t _wymix(uint64_t A, uint64_t B){ _wymum(&A,&B); return A^B; }

24static inline uint64_t _wyr4(const uint8_t *p) { uint32_t v; memcpy(&v, p, 4); return v;}

27//wyhash main function
28static inline uint64_t wyhash(const void *key, size_t len, uint64_t seed, const uint64_t *secret){
29 const uint8_t *p=(const uint8_t *)key; seed^=*secret; uint64_t a, b;
30 if(__builtin_expect(len<=16,true)){
31 if(__builtin_expect(len>=4,true)){ a=(_wyr4(p)<<32)|_wyr4(p+((len>>3)<<2)); b=(_wyr4(p+len-4)<<32)|_wyr4(p+len-4-((len>>3)<<2)); }
32 else if(__builtin_expect(len>0,true)){ a=_wyr3(p,len); b=0;}
33 else a=b=0;
34 }
35 else{
36 size_t i=len;
37 if(__builtin_expect(i>48,false)){
38 uint64_t see1=seed, see2=seed;
39 do{
40 seed=_wymix(_wyr8(p)^secret[1],_wyr8(p+8)^seed);
41 see1=_wymix(_wyr8(p+16)^secret[2],_wyr8(p+24)^see1);
42 see2=_wymix(_wyr8(p+32)^secret[3],_wyr8(p+40)^see2);
43 p+=48; i-=48;
44 }while(__builtin_expect(i>48,true));
45 seed^=see1^see2;
46 }
47 while(__builtin_expect(i>16,false)){ seed=_wymix(_wyr8(p)^secret[1],_wyr8(p+8)^seed); i-=16; p+=16; }
48 a=_wyr8(p+i-16); b=_wyr8(p+i-8);
49 }
50 return _wymix(secret[1]^len,_wymix(a^secret[1],b^seed));
51}
